SELECT S.SUBMISSION_ID, S.OFFER_ID, S.APPLICANT_STATUS,S.SUBMISSION_STATUS, TO_CHAR(S.LAST_UPDATE_DATE, 'DD/Mon/YY HH:MM:SS AM')FROM OBDX_SCHEMA.DIGX_OR_SUBMISSION_SUMMARY S 
WHERE OFFER_ID NOT IN ('NCS001','NCS006','NCS007')AND S.SUBMISSION_STATUS = 'DEDUPE'AND LAST_UPDATE_DATE >= '12-Oct-2017' ORDER BY LAST_UPDATE_DATE DESC
