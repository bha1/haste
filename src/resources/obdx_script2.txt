	TO_CHAR(S.LAST_UPDATE_DATE, 'DD/MM/YY HH:Mi:SS AM')
	FROM OBDX_PRD.DIGX_OR_SUBMISSION_SUMMARY S WHERE S.OFFER_ID  IN ('NCS001','NCS006','NCS007')
	AND S.APPLICANT_STATUS= 'EXISTING'
	AND S.SUBMISSION_STATUS IS NOT NULL
	AND S.LAST_UPDATE_DATE >= ?
	ORDER BY S.LAST_UPDATE_DATE DESC
